# vue-router 路由基本配置 与 注册路径

  • 安装：在安装 vue-cli 时选择自动安装
  • 引用：在主入口 main.js 中import router from "./router";引入， vue-cli 会自动配置好
  • 配置路由文件，vue-cli 会自动配置好
  • 两种引入组件的方法
// 方法一
import Home from "../views/Home.vue";
// 方法二
() => import(/* webpackChunkName: "about" */ "../views/About.vue");

// 配置路由文件，注册路径

import Vue from "vue";
import VueRouter from "vue-router";
import Home from "../views/Home.vue";

Vue.use(VueRouter);

const routes = [
  {
    path: "/asaaaa", //指定要跳转的路径
    name: "Home",
    //指定要跳转的路径对应的组件，组件需要引入
    component: {
      main: Home,
    },
  },
  {
    path: "/about",
    name: "About",
    // route level code-splitting 路由等级代码拆分
    // this generates a separate chunk (about.[hash].js) for this route which is lazy-loaded when the route is visited.
    // 为 当访问时延迟加载的路由 单独生成一个chunk
    component: () =>
      import(/* webpackChunkName: "about" */ "../views/About.vue"),
  },
];

const router = new VueRouter({
  mode: "history",
  base: process.env.BASE_URL,
  routes, //这里相当于routes:[前面声明的routs数组]
});

export default router;

# vue-router 路由传递参数

  1. 必须在路由内加入路由的 name

  2. 必须在 path 后加 /:要传递的参数

  3. 方案一(推荐)

  • 传递参数 <router-link :to="{name:'name',params:{要传递的参数：传递值}}"></router-link>
  • 在组件 html 标签内接收参数$route.params.要传递的参数
  • 或在组件实例内接收参数this.$route.params.要传递的参数
  • 参数以 url 路径……/Home/传递值的形式传递
  1. 方案二
  • 传递参数 <router-link :to="{path:'/name',query:{要传递的参数：传递值}}"></router-link>
  • 在组件中接收参数见案例
  • 参数以路径上添加查询参数的形式传递……/Home/?name=xxx&yyy=zzz …
  • 案例：知乎

# 案例

const routes = [
  {
    name: "Home",
    // path路径后添加查询参数
    path: "/Home/:homemsg",
    component: Home,
  },
];

# 方案一(推荐)

<!-- app.vue中传递参数 -->
<template>
  …
  <router-link
    :to="{
      // to前面要加冒号：
      name: "Home", // name与路由注册的name一致
      params: { homemsg: '这是传递的数据' }, // params是个对象，{要传递的参数：传递值}
      // 点击这个路由，是切换组件，所以这个数据是要跳转的组件接收
    }"
  >
    Home
  </router-link>
  …
</template>

<!-- 组件home.vue中接收参数 -->
<template>
  <div class="home">
    <div>
      {{ $route.params.homemsg }}
    </div>
  </div>
</template>

# 方案二

<!-- app.vue中传递参数 -->
<template>
  …
  <router-link
    :to="{
      path: "/Home",
      query: { homemsg: '这是传递的数据' },
    }"
  >
    Home
  </router-link>
  …
</template>

<!-- 函数模式读取参数 -->
<template>
  <div class="home">
    <div>
      {{ $route.params.homemsg }}
    </div>
  </div>
</template>

<script>
  // 创建一个函数返回props，以便可以将参数转换成另一种类型，将静态值与基于路由的值结合等。
  const router = new VueRouter({
    routers:[{
      path:"/Home",
      component:Home,
      props:(route)=>({query:route.query.homemsg})
    }]
  })
</script>
