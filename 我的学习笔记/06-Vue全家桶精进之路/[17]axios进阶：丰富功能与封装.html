# 初级版本

import Vue from "vue";
//1.import引入插件
import Axios from "axios";
//2.生成实例
const axios = Axios.create({});
//3.添加请求拦截器
axios.interceptors.request.use(
  // 请求之前做什么
  (req) => {
    //设置请求头
    // req.headers = { ...req.headers, a: 1 };
    return req;
  },
  (error) => {
    // 对请求错误做些什么
    console.log(error);
    return Promise.reject(error);
  }
);
// 添加响应拦截器，在真正返回response之前使用
axios.interceptors.response.use(
  (response) => {
    // 对响应数据做点什么
    return response;
  },
  (error) => {
    // 对响应错误做点什么
    return Promise.reject(error);
  }
);
//4.将请求封装成函数
const $axios = (url, data) => {
  //axios的生命周期放大了一圈
  if (url.indexOf("cnode") > -1) {
    return axios(url, data).then((res) => res.data);
  } else {
    return Promise.resolve();
  }
};
//5.直接调用方法
$axios("https://cnodejs.org/api/v1/topic/" + "5f322ce36c9ee864cf0e4751").then(
  (res) => {
    console.log(res);
  },
  (err) => {
    console.log(err);
  }
);
//6.或者添加到vue原型链中方便在组件中调用
Vue.prototype.$http = $axios;
//或者抛出，在组件中引入使用
export default $axios;

# config 解读与规划

  • axios 的 config 配置参数采用的是覆盖方式，如果你 created 创建实例时设置了值就覆盖

  • 如果没有设置值，也会有这个配置参数，采用的是 config 对象的默认值

  • ❀ 固定选项

{
  // 除非`url`是绝对的，否则`baseURL`将被加在`url`之前
  baseURL: 'https://some-domain.com/api/',
  // `headers` 设置请求头
  headers: {'X-Requested-With': 'XMLHttpRequest'},
  // `timeout` 指定请求超时之前的毫秒数
  // 如果超时，请求将被中止
  timeout: 1000, // 默认为0（无超时）
  // `withCredentials`指示是否应使用凭证进行跨站点访问控制请求，即请求是否携带cookie
  withCredentials: false, // default
}

  • ❀ 动态的，不同请求将有不同的设置，暴露出参数接口
{
  // 请求服务器的URL
  url: '/user',
  // 使用的请求方法
  method: 'get', // default
  // `params` 是要与请求一起发送的URL参数，必须是普通对象或URLSearchParams对象
  params: {
    ID: 12345
  },
  // `data` 是要作为请求正文发送的数据
  // 仅适用于请求方法'PUT'，'POST'，'DELETE和'PATCH'
  // 未设置`transformRequest`时，必须为以下类型之一：
  // - 字符串，纯对象，ArrayBuffer，ArrayBufferView，URLSearchParams
  // - 仅浏览器：FormData，File，Blob
  // - 仅node：Stream，Buffer
  data: {
    firstName: 'Fred'
  },
  // `cancelToken` 指定可用于取消请求的取消令牌cancel token
  // （请参见下面的“取消”部分以了解详细信息）
  cancelToken: new CancelToken(function (cancel) {
  }),
}

  • ❀ 暂时不用/少用，不设置，使用默认值，要用再特殊改造
{
  // `transformRequest`允许在将请求数据发送到服务器之前对其进行更改
  // 仅适用于请求方法'PUT'，'POST'，'PATCH'和'DELETE'
  // 数组中的最后一个函数必须返回字符串或Buffer，ArrayBuffer，FormData或Stream 的实例
  // 可以修改headers对象
  transformRequest: [function (data, headers) {
    // 修改要转换数据
    return data;
  }],
  // `transformResponse`允许更改响应数据，然后将其传递给then / catch
  transformResponse: [function (data) {
    return data;
  }],
  // `paramsSerializer` is an optional function in charge of serializing `params`
  // (例如 https://www.npmjs.com/package/qs, http://api.jquery.com/jquery.param/)
  paramsSerializer: function (params) {
    return Qs.stringify(params, {arrayFormat: 'brackets'})
  },
  // syntax alternative to send data into the body
  // method post
  // 仅发送值，而不发送键
  data: 'Country=Brasil&City=Belo Horizonte',
  // `adapter` 允许自定义处理请求，这使得测试更加容易。
  // 返回一个promise 且提供一个有效的response (see lib/adapters/README.md).
  adapter: function (config) {
    /* ... */
  },
  // `auth` 指应使用HTTP Basic auth，并提供凭据。
  // 将会设置一个`Authorization`的请求头, 覆盖现有`headers`的设置
  // `Authorization` custom headers you have set using `headers`.
  // `Authorization`自定义请求头，只能通过此参数配置HTTP Basic身份验证
  // 对于Bearer令牌等，请改用`Authorization`自定义请求头
  auth: {
    username: 'janedoe',
    password: 's00pers3cret'
  },
  // `responseType` 表示数据与服务器将响应类型
  // 选项有： 'arraybuffer', 'document', 'json', 'text', 'stream'
  // 浏览器只有: 'blob'
  responseType: 'json', // default
  // `responseEncoding`表示用于解码响应的编码（仅适用于Node.js）
  // 注意：忽略了“stream”或客户端请求的“responseType”
  responseEncoding: 'utf8', // default
  // `xsrfCookieName`是cookie的名称，用作xsrf令牌的值
  xsrfCookieName: 'XSRF-TOKEN', // default
  // `xsrfHeaderName`是带有xsrf令牌值
  xsrfHeaderName: 'X-XSRF-TOKEN', // default
  // `onUploadProgress` 允许处理上载的进度事件，仅适用于浏览器
  onUploadProgress: function (progressEvent) {
    // Do whatever you want with the native progress event
  },
  // `onDownloadProgress`允许处理下载的进度事件，仅适用于浏览器
  onDownloadProgress: function (progressEvent) {
    // Do whatever you want with the native progress event
  },
  // `maxContentLength`定义http响应内容的最大大小，以node.js中允许的字节数为准
  maxContentLength: 2000,
  // `maxBodyLength`（仅node选项）定义http请求内容的最大大小（以字节为单位）
  maxBodyLength: 2000,
  // `validateStatus`对HTTP响应状态代码解析还是拒绝承诺。
  // 如果`validateStatus`返回`true`（或设置为`null` //或`undefined`），则承诺将得到解决；否则，诺言将被拒绝。
  validateStatus: function (status) {
    return status >= 200 && status < 300; // default
  },
  // `maxRedirects` 定义在node.js中要遵循的最大重定向数。
  maxRedirects: 5, // default
  // `socketPath`定义要在node.js中使用的UNIX Socket
  // e.g. 'ar/run/docker.sock' 将请求发送到docker守护进程
  // 只能指定`socketPath`或`proxy`，如果两者都指定，则使用`socketPath`
  socketPath: null, // default
  // `httpAgent` and `httpsAgent` 定义在node.js中分别执行http：//和https请求时要使用的自定义代理。这样可以添加默认情况下未启用的选项. 例如`keepAlive`
  httpAgent: new http.Agent({ keepAlive: true }),
  httpsAgent: new https.Agent({ keepAlive: true }),
  // `proxy` 定义代理服务器的主机名，端口和协议。
  // 您还可以使用常规的`http_proxy`和`https_proxy`环境变量来定义代理。
  // 如果您使用环境变量作为代理配置，还可以定义一个`no_proxy`环境变量作为不被代理的comma-separated列表的domains。
  // 使用`false`禁用代理，忽略环境变量。
  // `auth`指示应使用HTTP Basic auth连接到代理，并提供凭据。
  // 这将设置一个“Proxy-Authorization”自定义请求头，覆盖现有的“headers”设置
  // 如果代理服务器使用HTTPS，则必须将协议设置为`https`
  proxy: {
    protocol: 'https',
    host: '127.0.0.1',
    port: 9000,
    auth: {
      username: 'mikeymike',
      password: 'rapunz3l'
    }
  },
  // `decompress` 指示是否应自动对响应主体进行解压缩
  //如果设置为“ true”,将会从所有解压缩响应的响应对象中删除'content-encoding'请求头
  // - 仅node（XHR无法关闭解压缩）
  decompress: true // default
}
