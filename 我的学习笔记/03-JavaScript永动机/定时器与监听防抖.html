# setTimeout()

setTimeout(function () {}, 1000);
// 等同于 具名函数fn
function fn() {}
setTimeout(fn, 1000);
// 等同于匿名函数
const obj = {
  y: function () {},
};
setTimeout(function () {
  obj.y();
}, 1000);

# setInterval()

var timer = setInterval(function () {
  console.log(2);
}, 1000);
1
2
3

  • setInterval()用于设定每隔 xxx 毫秒执行一次某个函数。

  • 给该计时器一个名字，可以实现某个条件时使用clearInterval()砸掉该计时器。

  • setTimeout 和 setInterval 函数，都返回一个整数值，表示计数器编号。将该整数传入 clearTimeout 和 clearInterval 函数，就可以取消对应的定时器。

# 事件监听防抖

let timer = null
function () {
  window.addEventListener(
    "scroll", () => {
      // 防抖
      if (timer) {
        clearTimeout(timer)
      }
      // timer还没到时间点=undefined，或第一次为null都为假不砸，
      // 到点了得到返回值timer=222为真，就把所有计时器砸了，
      // 此时，还在undefined还没到执行时间的计时器通通砸掉清零，
      // 因为监听一直在，重新造了个计时器重新开始。实现每隔100毫秒执行监听函数一次。
      timer = setTimeout(() => {
        XXXXX
      }, 100)
      // 每次监听产生一个计时器(闹钟)100毫秒后执行1次监听函数
      // 还没到100毫秒未执行timer=undefined;执行timer=计数器编号222，
      // 由于监听一次就会产生一个闹钟，同一个时间会存在很多个闹钟，且定时不同。
    }
  )
}.call()

# 时间参数

const fn = function () {};
const duration = 1000;
const timer = setInterval(fn, duration);
1
2
3
  • 计时器一旦生成，只在生成时读取 1 次时间间隔参数 duration，以后固定每隔该时间执行一次。因此修改时间参数 duration 也无法改变该计时器的时间间隔。
  • 解决方案：
let n = 0;
const duration = 1000;
let id = setTimeout(function fn() {
  n += 1;
  if(n条件，计时器还未结束/需要修改时间间隔的情况){
    id = setTimeout(fn , duration)
  }
},duration)
// 实现了每隔duration调用一次经过duration时间后再次执行fn的目的
// 达到2个目标：
// 1.模拟setInterval时间间隔；
// 2.每次执行setTimeout(fn , duration)都能重新调用1次duration当前值
