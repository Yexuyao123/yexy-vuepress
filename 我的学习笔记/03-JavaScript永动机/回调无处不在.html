# MVC 中的回调

  • MVC 中
  • obj 对象里的 init 是大家共用的,及每个调用的模块都需要初始化的内容
  • 而通过 options 对象传参传进来的 init 是每个模块需要特殊初始化的内容
  • 这里的 options 就是回调
const Controller = function (options) {
  const initFromOptions = options.init;
  const obj = {
    view: null,
    model: null,
    init: function (view, model) {
      //这个init是对象obj的属性,属性值为函数
      console.log("obj init run");
      this.view = view;
      this.model = model;
      this.model.init();
      initFromOptions.call(this, view, model);
      // 这里的this是init函数里面的this
      // 它指向调用init函数的上下文,即obj
      // 同时XXX.call()第一个参数会重定向XXX函数里面的this
      // XXX函数也就是有options传进来的initFromOptions即options.init
      // 所以options.init中的this全部是obj,
      // 所以.call(this, view, model)中的view和model是通过init已经初始化得到的view和model
      options.bindEvents.call(this);
    },
  };

  for (let key in options) {
    if (key === "init" || key === "bindEvents") {
    } else {
      obj[key] = options[key];
    }
  }

  return obj;
};

const options = {
  a: 1,
  b: 2,
  init: function (view, model) {
    console.log("options init run");
  },
  bindEvents: function () {
    console.log("bindEvents run");
  },
};
let view = document.documentElement;
let model = {
  init: function () {
    console.log("model init run");
  },
};

let c = Controller(options);
c.init(view, model);

# 异步中的回调

  • 回调可以是异步的也可以是同步的

  • 异步任务是那些被引擎放在一边，不进入主线程、而进入任务队列的任务。

  • 只有引擎认为某个异步任务可以执行了（比如 Ajax 操作从服务器得到了结果），该任务（采用回调函数的形式）才会进入主线程执行。

  • 不会阻塞主线程

  • 排在异步任务后面的代码，无需等待异步任务结束会马上运行，也就是说，异步任务不具有“堵塞”效应。

  • JavaScript 中的异步代码发展历程：

  1. 回调：
  • 简单的回调函数：当 xxx 条件发生时调用某个函数；
  • 通过事件监听、if 条件语句、while/for 循环 实现。
  1. Promise 对象

  • Promise 对象用于表示一个异步操作的最终完成 (或失败)及其结果值，把异步操作最终的成功返回值或者失败原因和相应的处理程序关联起来。

  • 一个 Promise 必然处于以下几种状态之一：

    • 待定（pending）: 初始状态，既没有被兑现，也没有被拒绝。
    • 已兑现（fulfilled）: 意味着操作成功完成。
    • 已拒绝（rejected）: 意味着操作失败。
const fn = function (XXX) {
  return new Promise((resolve, reject) => {
    if (XXX) {
      resolve("存在");
    } else {
      reject("不存在");
    }
  });
};
fn(XXX).then(
  (resolve) => {
    console.log(resolve);
  }, // "存在"
  (reject) => {
    console.log(reject);
  } // "不存在"
);
