# 4.使用 JS SDK 安装 LC

# 方案 1：npm 安装

$ npm install leancloud-storage --save

# 方案 2：CDN——通过 script 标签引入

  • <script src="//cdn.jsdelivr.net/npm/leancloud-storage@4.9.0/dist/av-min.js"></script>

# 5.初始化+测试

# 初始化

AV.init({
  appId: "xxxxxxxxxx",
  appKey: "xxxxxxxxxx",
  serverURL: "https://please-replace-with-your-customized.domain.com",
});
console.log("初始化成功");

注意：

  • 不同账户不同项目下的 appId 和 appKey 不同，需要选中当前的项目，并在账户中生成。
  • 请将 https://please-replace-with-your-customized.domain.com 替换为你的应用绑定的 API 域名。
  • 如果没有域名，可以在 当前项目 → 设置 → 应用 keys 中找到 [ REST API 服务器地址 ] ，复制替换到 serverURL 中。这是个共享域名，不能保证可用性，可临时使用该地址进行开发，在正式发布前为当前应用绑定 API 访问域名。

# 测试

确认本地网络环境是可以访问 LeanCloud 服务器的，可以执行以下命令： curl "https://API_BASE_URL/1.1/date"

然后在项目中编写如下测试代码：

const TestObject = AV.Object.extend("TestObject");
const testObject = new TestObject();
testObject.set("words", "Hello world!");
testObject.save().then((testObject) => {
  console.log("保存成功。");
});

# .筛选方法()

query.equalTo("lastName", "Smith"); // 查询所有 lastName 为 Smith 的对象
query.notEqualTo("firstName", "Jack"); // 查询所有 firstName 不为 Jack 的对象
query.exists("images"); // 查找包含 'images' 的对象
query.doesNotExist("images"); // 查找不包含 'images' 的对象
// 对于能够排序的属性（比如数字、字符串），可以进行比较查询：
query.lessThan("age", 18); // 限制 age < 18
query.lessThanOrEqualTo("age", 18); // 限制 age <= 18
query.greaterThan("age", 18); // 限制 age > 18
query.greaterThanOrEqualTo("age", 18); // 限制 age >= 18
query.limit(10); // 最多获取 10 条结果
query.skip(20); // 跳过前 20 条结果
query.ascending("createdAt"); // 按 createdAt 升序排列
query.descending("createdAt"); // 按 createdAt 降序排列
query.addAscending("priority"); //设置排序规则,按照 priority 排序
query.greaterThanOrEqualTo("balance", amount); // balance 值大于或等于 amount 值

// 用户的账户表 Account 有一个余额字段 balance，同时有多个请求要修改该字段值。为避免余额出现负值，只有当金额小于或等于余额的时候才能接受请求：
const account = AV.Object.createWithoutData(
  "Account",
  "5745557f71cfe40068c6abe0"
);
// 对 balance 原子减少 100
const amount = -100; // 交易金额100即账户余额-100
account.increment("balance", amount);
account
  .save(null, {
    // 设置条件
    query: new AV.Query("Account").greaterThanOrEqualTo("balance", -amount),
    // 操作结束后，返回最新数据。
    // 如果是新对象，则所有属性都会被返回，
    // 否则只有更新的属性会被返回。
    fetchWhenSave: true,
  })
  .then(
    (account) => {
      console.log(`当前余额为：${account.get("balance")}`);
    },
    (error) => {
      if (error.code === 305) {
        console.error("余额不足，操作失败！");
      }
    }
  );

# 删除(destroy/unset)

下面的代码从云端删除一个 Todo 对象；

const todo = AV.Object.createWithoutData("Todo", "582570f38ac247004f39c24b");
todo.destroy();
1
2

如果只需删除对象的一个属性，可以用 unset：

// priority 属性会被删除
todo.unset("priority");
1
2

删除对象是一个较为敏感的操作，我们建议你阅读 ACL 权限管理开发指南 来了解潜在的风险。熟悉 class 级别、对象级别和字段级别的权限可以帮助你有效阻止未经授权的操作。
