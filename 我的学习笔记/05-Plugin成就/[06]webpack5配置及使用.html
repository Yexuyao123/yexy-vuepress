# ① 主入口 index.js 的打包

入口起点(entry point)指示 webpack 应该使用哪个模块，来作为构建其内部 依赖图(dependency graph) 的开始。 进入入口起点后，webpack 会找出有哪些模块和库是入口起点（直接和间接）依赖的。 默认值是 ./src/index.js，但你可以通过在 webpack configuration 中配置 entry 属性，来指定一个（或多个）不同的入口起点。

目的：通过配置将脚本 src/index.js 作为 入口起点，生成 dist/main.js 作为 输出。

# 主入口 index.js 引入 lodash

// src/index.js中写入
import _ from "lodash";
1
2

# 配置 webpack.config.js

配置将脚本 src/index.js 作为 入口起点，生成 dist/main.js 作为 输出。

// webpack.config.js配置
const path = require("path");

module.exports = {
  entry: "./src/index.js",
  output: {
    filename: "./main.js",
    path: path.resolve(__dirname, "dist"), // 设置路径当前位置文件夹/dist, dirname是运行命令行所在的当前位置
  },
};

# ②babel-loader(处理 src 全部 js 文档的 ES 版本)

  • babel-loader——webpack 官方文档说明 (opens new window)

  • Babel 是一个工具链，主要用于将 ECMAScript 2015+ 版本的代码转换为向后兼容的 JavaScript 语法，以便能够运行在当前和旧版本的浏览器或其他环境中。

  • bbabel-loader v8 + Babel v7——github (opens new window)

  • babel 官方中文文档 (opens new window)

  • 注意：babel-loader 和 babel 是有版本搭配要求的

  • babel-loader v8 + Babel v7 或者 babel-loader v7 + Babel v6

  • Babel v6 请查看babel-loader v7——github (opens new window)

  • 下面介绍 babel-loader v8 + Babel v7 方案的安装与配置

# babel-loader

# 安装
npm install -D babel-loader @babel/core @babel/preset-env
1

如果一直报错，就将三个插件分开一个个安装。

# 配置
// webpack.config.js配置
module: {
  output: {
    //(可选) 目标环境支持某些ES2015功能
    environment: {
      arrowFunction: true,
    },
  },
  rules: [
    {
      test: /\.m?js$/,
      exclude: /node_modules/, //排除node_modules中的库(不错处理)
      not: [ //(可选) 排除应避开node_modules中的XXX库(要处理)
        // /除了其中一些因使用现代语法而需要翻译的库之外
        /unfetch/,
        /d3-array|d3-scale/,
        /@hapi[\\/]joi-date/,
      ]
      use: {
        loader: "babel-loader",
        options: {
          presets: [["@babel/preset-env", { targets: "defaults" }]],
            // 也可单独指定targets: "ie 11"
          plugins: ["@babel/plugin-proposal-class-properties"],
        },
      },
    },
  ];
}

# 引入 Babel runtime 作为一个独立模块，来避免重复引入

  • 下面的配置禁用了 Babel 自动对每个文件的 runtime 注入，而是引入 @babel/plugin-transform-runtime 插件并且使所有辅助代码从这里引用。
  • 并且把 @babel/runtime 安装为一个依赖。
# 安装
npm install -D @babel/plugin-transform-runtime
npm install @babel/runtime
1
2
# 配置
// webpack.config.js配置
use: {
        loader: "babel-loader",
        options: {
          presets: [["@babel/preset-env", { targets: "defaults" }]],
          plugins: ["@babel/plugin-transform-runtime"],
          // 'transform-runtime'插件告诉Babel 需要运行而不是对其进行内联。
        },
      },

# css 文档 打包 mini-css-extract-plugin

  • 使用 style-loader 会将 css 内容以插入 style 标签的方式打包到输出的 html 中。

  • 如果希望 css 文档单独打包并且以 link 标签的形式引入输出的 html 中，应使用 mini-css-extract-plugin 插件。

  • 注意：style-loader 和 ini-css-extract-plugin 两种方案只能二选一。

  • mini-css-extract-plugin——npm 文档 (opens new window)

  • mini-css-extract-plugin——github 文档 (opens new window)

# 安装

npm install --save-dev mini-css-extract-plugin
1

# 配置

//index.js中import要输出的css/scss
import "./style.scss";
1
2
// webpack.config.js配置
const MiniCssExtractPlugin = require("mini-css-extract-plugin");
module.exports = {
  optimization: {
    //将所有CSS提取到一个文件中
    splitChunks: {
      cacheGroups: {
        styles: {
          name: "styles", //输出的css文件名
          type: "css/mini-extract", //路径是dist输出文件夹下的css文件夹
          // For webpack@4
          // test: /\.css$/,
          chunks: "all",
          enforce: true,
        },
      },
    },
  },
  module: {
    rules: [
      {
        test: /\.css$/i,
        use: [
          MiniCssExtractPlugin.loader,
          "css-loader",
          "postcss-loader",
          "sass-loader",
        ],
      },
    ],
  },
  plugins: [
    new MiniCssExtractPlugin({
      filename: "css/[name].css", //此选项确定每个输出CSS文件的名称 //打包同步代码
      chunkFilename: "css/[id].css", // 此选项确定碎片块文件的名称。//打包异步代码
      // 此选项在dom元素前插入CSS的link标签
      // insert: function (linkTag) {
      //   var reference = document.querySelector('#title');
      //   if (reference) {
      //     reference.parentNode.insertBefore(linkTag, reference);
      //   }
      // },
      insert: "#title", //<link>将在id所指元素之后插入一个新元素
    }),
  ],
};
